def somma(float a, b): float
{
	return a+b;
}


def differenza(float a | float b): float
{
	return a-b;
}


def prodotto(float a, b): float
{
	return a*b;
}


def divisione(float a, b): float
{
	return a/b;
}

def potenza(float a, b, c): float
{

	return a^b;
}

def succ_fibonacci(integer i): integer {

    if i<0 then{
        return 0;
    }else {
        if i=0 then {
            return 0;
        }
        else
        {
            if i=1 then {
                return 1;
            }


        }
    }

    return succ_fibonacci(i-1)+succ_fibonacci(i-2);
}
