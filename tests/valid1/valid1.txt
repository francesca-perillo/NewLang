def sommac(integer a, d | float b | out string size): float
{
	float result;


	result  <<  a + b + c + d;

	if result > 100 then{
		var valore << "grande";
 		size << valore; }
	else {
		var valore << "piccola";
 		size << valore;
	}

	return result;
}
